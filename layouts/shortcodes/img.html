{{ if $image }} {{- $webp400 := $image.Resize "400x webp" -}} {{- $webp800 := $image.Resize "800x webp" -}} {{- $webp1200 := $image.Resize "1200x webp" -}} {{- $jpg400 := $image.Resize "400x jpg" -}} {{- $jpg800 := $image.Resize "800x jpg" -}} {{ else }}
